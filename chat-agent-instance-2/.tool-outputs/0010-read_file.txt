import { chromium } from 'playwright';

(async () => {
  console.log('🔍 正在启动浏览器并加载登录状态...');
  
  const browser = await chromium.launch({
    headless: false
  });
  
  const context = await browser.newContext({
    storageState: 'weibo-auth.json'
  });
  
  const page = await context.newPage();
  
  try {
    // 访问用户主页
    console.log('🔗 正在访问用户主页: https://weibo.com/u/1924675227');
    await page.goto('https://weibo.com/u/1924675227', { timeout: 60000 });
    await page.waitForTimeout(8000);  // 等待页面完全加载
    
    console.log('📄 当前 URL:', page.url());
    
    // 截图
    await page.screenshot({ path: 'weibo-user-page-full.png', fullPage: true });
    console.log('📸 全页面截图已保存');
    
    // 提取页面所有可见文本
    const allText = await page.evaluate(() => {
      // 获取页面主体内容
      const body = document.body;
      return body.innerText;
    });
    
    console.log('\n📝 页面文本内容:\n');
    console.log('='.repeat(80));
    console.log(allText.substring(0, 3000));  // 打印前3000字符
    console.log('='.repeat(80));
    
    // 保存完整文本
    const fs = await import('fs');
    fs.writeFileSync('weibo-user-page-content.txt', allText, 'utf-8');
    console.log('\n💾 完整页面文本已保存到 weibo-user-page-content.txt');
    
    // 尝试提取特定信息
    const userInfo = await page.evaluate(() => {
      // 查找用户名
      const usernameEl = document.querySelector('[class*="username"]') || 
                          document.querySelector('[class*="name"]') ||
                          document.querySelector('h1');
      
      // 查找简介
      const descEl = document.querySelector('[class*="description"]') ||
                     document.querySelector('[class*="bio"]') ||
                     document.querySelector('[class*="intro"]');
      
      // 查找位置
      const locationEl = document.querySelector('[class*="location"]');
      
      // 查找粉丝数
      const followersEl = document.querySelector('[class*="follower"]') ||
                          document.querySelector('[node-type*="follower"]');
      
      // 查找关注数
      const followingEl = document.querySelector('[class*="follow"]') ||
                          document.querySelector('[node-type*="follow"]');
      
      // 查找微博内容
      const weiboPosts = Array.from(document.querySelectorAll('[class*="WB_card"] [class*="WB_text"]'))
        .slice(0, 5)
        .map(el => el.innerText);
      
      return {
        username: usernameEl?.innerText || '',
        description: descEl?.innerText || '',
        location: locationEl?.innerText || '',
        followers: followersEl?.innerText || '',
        following: followingEl?.innerText || '',
        recentPosts: weiboPosts
      };
    });
    
    console.log('\n📊 提取的用户信息:');
    console.log('  用户名:', userInfo.username || '未找到');
    console.log('  简介:', userInfo.description || '未找到');
    console.log('  位置:', userInfo.location || '未找到');
    console.log('  粉丝:', userInfo.followers || '未找到');
    console.log('  关注:', userInfo.following || '未找到');
    console.log('  最近微博:', userInfo.recentPosts.length > 0 ? userInfo.recentPosts.length + '条' : '未找到');
    
    // 保存用户信息
    const fs = await import('fs');
    fs.writeFileSync('weibo-user-info-extracted.json', JSON.stringify(userInfo, null, 2), 'utf-8');
    console.log('\n💾 用户信息已保存到 weibo-user-info-extracted.json');
    
    console.log('\n⏳ 等待 15 秒供查看...');
    await page.waitForTimeout(15000);
    
  } catch (error) {
    console.error('❌ 错误:', error.message);
    await page.screenshot({ path: 'weibo-extract-error.png' });
  } finally {
    await browser.close();
  }
})();
